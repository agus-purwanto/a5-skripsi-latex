\end{tabular}

\item Pepetkan garis horizontalnya \\
\begin{tabular}{@{}rrrr@{}}\hline

122 & 8 &  2& 3\\

\end{tabular}

122 & 878 &  2& 3\\ \bottomrule
\end{tabular}
